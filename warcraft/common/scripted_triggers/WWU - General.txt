has_river_estuary_trigger = {
	custom_trigger_tooltip = {
		tooltip = has_river_estuary_tooltip
	}
}

is_great_leader = {
	OR = {
		AND = {
			adm = 5
			dip = 5
		}
		AND = {
			adm = 5
			mil = 5
		}
		AND = {
			dip = 5
			mil = 5
		}
	}
}

can_form_order_for_class = {
	NOT = { 
		has_country_flag = has_$class$_order 
	} 
	NOT = { 
		any_country = { 
			has_country_flag = has_$class$_order 
		} 
	}
}

highest_development = {
	NOT = { any_country = {
		NOT = { tag = ROOT }
		total_development = ROOT
	} }
}

owns_or_vassal_or_ally_of = {
	OR = {
		$country$ = {
			owns = $province$
		}
		$country$ = {
			any_subject_country = {
				owns = $province$
			}
		}
		$country$ = {
			any_ally = {
				owns = $province$
			}
		}
	}
}

ally_or_subject_or_overlord_with = {
	$who$ = {
		OR = {
			is_subject_of = $with$
			overlord_of = $with$
			alliance_with = $with$
		}
	}
}

was_at_same_side_of_war = {
	OR = {
		any_country = {
			$who$ = {
				has_country_flag = declared_war_to@PREV
			}
			$with$ = {
				has_country_flag = declared_war_to@PREV
			}
		}
		any_country = {
			has_country_flag = declared_war_to@$who$
			has_country_flag = declared_war_to@$with$
		}
	}
}

was_at_same_side_of_war_against = {
	OR = {
		AND = {
			$who$ = {
				has_country_flag = declared_war_to@$against$
			}
			$with$ = {
				has_country_flag = declared_war_to@$against$
			}
			ally_or_subject_or_overlord_with = {
				who = $who$
				with = $with$
			}
		}
		AND = {
			$against$ = {
				has_country_flag = declared_war_to@$who$
			}
			$against$ = {
				has_country_flag = declared_war_to@$with$
			}
			ally_or_subject_or_overlord_with = {
				who = $who$
				with = $with$
			}
		}
	}
}

same_region = {
	$prov_1$ = {
		region_for_scope_province = {
			province_id = $prov_2$
		}
	}
}

has_been_insulted_by_root = {
	custom_trigger_tooltip = {
		tooltip = has_been_insulted_by_root
		OR = {
			has_opinion_modifier = {
				who = ROOT
				modifier = insulted
			}
			has_opinion_modifier = {
				who = ROOT
				modifier = scornfully_insulted
			}
		}
	}
}

	is_subject = no
